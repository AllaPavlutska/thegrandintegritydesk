class DatabaseManager:
    """Handles SQLite database lifecycle and persistence."""
    
    def __init__(self, db_name: str = "safety_pipeline.db"):
        self.db_name = db_name
        self._initialize_db()

    def _initialize_db(self) -> None:
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS detections (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    url TEXT NOT NULL,
                    article_title TEXT,
                    is_misinfo INTEGER,
                    confidence_score REAL,
                    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
                )
            ''')
            conn.commit()

    def log_detection(self, url: str, title: str, is_misinfo: int, confidence: float) -> None:
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                INSERT INTO detections (url, article_title, is_misinfo, confidence_score)
                VALUES (?, ?, ?, ?)
            ''', (url, title, int(is_misinfo), float(confidence)))
            conn.commit()


class MisinformationEngine:
    """Evaluates text content against domain reputation and heuristic risk patterns."""
    
    def __init__(self):
        # Merge lists for efficient O(1) lookup
        self.blacklisted_domains = GENERAL_BLACKLIST.union(RUSSIAN_STATE_MEDIA_BLACKLIST)
        self.trigger_phrases = TRIGGER_PHRASES

    def check_domain_rules(self, url: str) -> bool:
        """Checks if the URL contains any domain from the restricted lists."""
        url_lower = url.lower()
        return any(domain in url_lower for domain in self.blacklisted_domains)

    def extract_text(self, url: str) -> Tuple[str, str]:
        """Downloads and parses article content."""
        try:
            article = Article(url)
            article.download()
            article.parse()
            return article.title, article.text
        except Exception as e:
            raise ValueError(f"Content extraction failed: {str(e)}")

    def analyze(self, text: str) -> Tuple[int, float]:
        """Runs heuristic analysis on text content to determine misinformation probability."""
        if not text.strip():
            raise ValueError("The article appears to be empty or inaccessible.")
            
        text_lower = text.lower()
        trigger_hits = sum(1 for phrase in self.trigger_phrases if phrase in text_lower)
                
        if trigger_hits == 0:
            return 0, 0.95 
        elif trigger_hits == 1:
            return 1, 0.60 
        elif trigger_hits == 2:
            return 1, 0.80 
        else:
            return 1, 0.99
