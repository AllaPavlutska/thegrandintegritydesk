def validate_url(url: str) -> Tuple[bool, str]:
    """
    Validates a URL before any network request is made.
    Returns (is_valid, error_message).
    """
    url = url.strip()
    if not url:
        return False, "The Inspector requires a valid URL before proceedings may begin."

    if not url.startswith(("http://", "https://")):
        return False, "All submitted documents must bear a proper address — beginning with http:// or https://"

    try:
        parsed = urlparse(url)
        if not parsed.netloc or '.' not in parsed.netloc:
            return False, "The domain portion of this address appears to be missing or malformed."
        hostname = parsed.hostname or ""
        if hostname in ("localhost", "127.0.0.1", "0.0.0.0") \
                or hostname.startswith("192.168.") \
                or hostname.startswith("10."):
            return False, "Local and private network addresses are strictly forbidden by hotel policy."
        return True, ""
    except Exception:
        return False, "The address could not be parsed. Kindly verify the format and resubmit."

class DatabaseManager:
    """Handles SQLite database lifecycle and persistence."""

    def __init__(self, db_name: str = "safety_pipeline.db"):
        self.db_name = db_name
        self._initialize_db()

    def _initialize_db(self) -> None:
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS detections (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    url TEXT NOT NULL,
                    article_title TEXT,
                    is_misinfo INTEGER,
                    confidence_score REAL,
                    hit_count INTEGER DEFAULT 0,
                    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
                )
            ''')
            cursor.execute('CREATE INDEX IF NOT EXISTS idx_url ON detections(url)')
            cursor.execute('CREATE INDEX IF NOT EXISTS idx_timestamp ON detections(timestamp)')
            conn.commit()

    def log_detection(self, url: str, title: str, is_misinfo: int,
                      confidence: float, hit_count: int = 0) -> None:
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                INSERT INTO detections (url, article_title, is_misinfo, confidence_score, hit_count)
                VALUES (?, ?, ?, ?, ?)
            ''', (url, title, int(is_misinfo), float(confidence), hit_count))
            conn.commit()

    def get_recent_detections(self, limit: int = 15):
        """Fetch the most recent scan records for the registry view."""
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                SELECT url, article_title, is_misinfo, confidence_score, timestamp
                FROM detections
                ORDER BY timestamp DESC
                LIMIT ?
            ''', (limit,))
            return cursor.fetchall()

    def get_stats(self):
        """Return aggregate stats: total scans and flagged count."""
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('SELECT COUNT(*), SUM(is_misinfo) FROM detections')
            row = cursor.fetchone()
            total = row[0] or 0
            flagged = int(row[1] or 0)
            return total, flagged

# Heuristic Analysis Engine
# ---------------------------------------------------------------------------

class HeuristicEngine:
    """
    Evaluates text content against domain reputation and heuristic risk patterns.

    IMPORTANT: This is a rule-based heuristic system, NOT a machine learning model.
    Verdicts are based on keyword matching and domain blocklists.
    Results should be treated as a preliminary indicator, not a definitive fact-check.
    """

    def __init__(self):
        self.blacklisted_domains = GENERAL_BLACKLIST.union(RUSSIAN_STATE_MEDIA_BLACKLIST)
        self.trigger_phrases = TRIGGER_PHRASES

    def check_domain_rules(self, url: str) -> bool:
        """Returns True if the URL's domain is on a restricted list."""
        try:
            hostname = urlparse(url).hostname or ""
            return any(domain in hostname for domain in self.blacklisted_domains)
        except Exception:
            return False

    def extract_text(self, url: str) -> Tuple[str, str]:
        """
        Downloads and parses article content.
        Raises descriptive, thematically appropriate errors for specific failure modes.
        """
        try:
            article = Article(url)
            article.download()
            article.parse()
        except Exception as e:
            err = str(e).lower()
            if "404" in err or "not found" in err:
                raise ValueError(
                    "The document could not be located — the page returned a 404. "
                    "Please verify the address and resubmit."
                )
            elif "timeout" in err or "timed out" in err:
                raise ValueError(
                    "The Inspector's messenger has returned empty-handed. "
                    "The remote server did not respond in time."
                )
            elif "403" in err or "forbidden" in err:
                raise ValueError(
                    "Entry refused. The establishment has denied access — "
                    "this page may be paywalled or blocking automated visitors."
                )
            elif "ssl" in err or "certificate" in err:
                raise ValueError(
                    "The document's credentials could not be verified. "
                    "An SSL certificate error was encountered."
                )
            else:
                raise ValueError(
                    f"The document retrieval has encountered an unforeseen complication: {str(e)}"
                )

        if not article.text or len(article.text.strip()) < 50:
            raise ValueError(
                "The retrieved document contains very little readable text. "
                "It may be rendered by JavaScript, concealed behind a paywall, or require a login."
            )

        return article.title or "Untitled Document", article.text

    def analyze(self, text: str) -> Tuple[int, float, int]:
        """
        Runs heuristic analysis on text content.
        Returns (is_misinfo, heuristic_score, hit_count).

        Scoring rationale (heuristic thresholds, not statistical probabilities):
          0 hits  -> Likely authentic        (score: 0.95)
          1 hit   -> Low suspicion, cleared  (score: 0.40)
          2 hits  -> Moderate suspicion      (score: 0.65)
          3+ hits -> High suspicion          (score: 0.85)
        """
        if not text.strip():
            raise ValueError("The article appears to be empty or inaccessible.")

        text_lower = text.lower()
        hit_count = sum(1 for phrase in self.trigger_phrases if phrase in text_lower)

        if hit_count == 0:
            return 0, 0.95, 0
        elif hit_count == 1:
            return 0, 0.40, hit_count  # A single phrase hit is insufficient to condemn
        elif hit_count == 2:
            return 1, 0.65, hit_count
        else:
            return 1, 0.85, hit_count

def render_verdict(prediction: int, confidence: float, hit_count: int,
                   title: str, text: str, is_domain_blocked: bool) -> None:
    """Renders the official result section of a scan."""
    st.divider()
    st.markdown(
        "<h3 style='text-align:center; color:#C25953; font-family:Georgia;'>"
        "— Official Telegram —"
        "</h3>",
        unsafe_allow_html=True
    )

    col1, col2, col3 = st.columns(3)
    with col1:
        verdict = "CONDEMNED" if prediction == 1 else "CLEARED"
        st.metric(label="Official Verdict", value=verdict)
    with col2:
        st.metric(label="Certainty Score", value=f"{confidence * 100:.0f}%")
    with col3:
        st.metric(label="Phrases Flagged", value=str(hit_count))

    if is_domain_blocked:
        st.warning(
            "NOTICE: This domain is restricted by the Grand Authority. "
            "Its publications are not permitted within these walls."
        )
    elif prediction == 1:
        st.error(
            f"NOTICE: The Inspector has identified {hit_count} suspicious passage(s) "
            "within the submitted document. This material is hereby flagged for further review."
        )
    else:
        st.success(
            "NOTICE: The Inspector has found no cause for alarm. "
            "This document has passed preliminary examination and may circulate freely."
        )

    st.markdown(
        "<div class='method-note'>"
        "<strong>Inspector's Note:</strong> This assessment is produced by rule-based heuristic "
        "analysis — keyword matching against a curated phrase list and domain registry. "
        "It is <em>not</em> a machine learning system. Scores are thresholds, not statistical "
        "probabilities. The Grand Authority recommends corroborating all findings with a "
        "qualified fact-checker before drawing conclusions."
        "</div>",
        unsafe_allow_html=True
    )

    with st.expander("Review Confiscated Payload"):
        st.markdown(f"**Document Title:** {title}")
        if text and text != "Content extraction bypassed due to domain reputation policy.":
            st.markdown("**Content Snippet (first 500 characters):**")
            st.code(text[:500] + ("..." if len(text) > 500 else ""), language=None)
        else:
            st.info("No content was extracted for this entry.")


def render_history(db_manager: DatabaseManager) -> None:
    """Renders the inspection registry tab."""
    total, flagged = db_manager.get_stats()

    col1, col2, col3 = st.columns(3)
    col1.metric("Documents Inspected", total)
    col2.metric("Condemned", flagged)
    col3.metric("Cleared", total - flagged)

    st.divider()
    rows = db_manager.get_recent_detections(limit=15)

    if not rows:
        st.info("The registry is empty. No inspections have been conducted.")
        return

    st.markdown("**Registry of Recent Inspections**")
    for url, article_title, is_misinfo, confidence, timestamp in rows:
        flag_class = "flag-condemned" if is_misinfo else "flag-cleared"
        flag_text  = "CONDEMNED"      if is_misinfo else "CLEARED"
        short_url  = url[:65] + "..." if len(url) > 65 else url
        st.markdown(
            f"<div class='history-row'>"
            f"<span class='{flag_class}'>[{flag_text}]</span> "
            f"<strong>{article_title or 'Untitled Document'}</strong><br>"
            f"<small>{short_url}</small> &nbsp;|&nbsp; "
            f"Certainty: {confidence * 100:.0f}% &nbsp;|&nbsp; {timestamp}"
            f"</div>",
            unsafe_allow_html=True
        )

def main():
    st.set_page_config(page_title="The Grand Information Hotel", layout="centered")
    inject_custom_css()

    @st.cache_resource
    def get_system_components():
        return HeuristicEngine(), DatabaseManager()

    engine, db = get_system_components()

    st.markdown("<h1>The Grand Integrity Desk</h1>", unsafe_allow_html=True)
    st.markdown(
        "<p class='subtitle'>Please submit the document for formal inspection.</p>",
        unsafe_allow_html=True
    )
    st.markdown(
        "<div class='disclaimer-banner'>"
        "<strong>Notice to Guests:</strong> The Grand Integrity Desk employs heuristic analysis only. "
        "Its findings are preliminary in nature and do not constitute a definitive verdict. "
        "The management accepts no liability for conclusions drawn without further verification."
        "</div>",
        unsafe_allow_html=True
    )

    tab_scan, tab_history = st.tabs(["Scan Article", "Inspection Registry"])

    # ---- SCAN TAB ----
    with tab_scan:
        url_input = st.text_input(
            "Target URL",
            placeholder="— INSERT DOCUMENT ADDRESS HERE —",
            label_visibility="collapsed"
        )

        if st.button("Submit for Review", use_container_width=True):
            is_valid, error_msg = validate_url(url_input)
            if not is_valid:
                st.error(error_msg)
                return

            with st.spinner("The Inspector is reviewing the document..."):
                try:
                    is_domain_blocked = engine.check_domain_rules(url_input)

                    if is_domain_blocked:
                        prediction, confidence, hit_count = 1, 1.0, 0
                        title = "Restricted Domain"
                        text  = "Content extraction bypassed due to domain reputation policy."
                    else:
                        title, text = engine.extract_text(url_input)
                        prediction, confidence, hit_count = engine.analyze(text)

                    db.log_detection(url_input, title, prediction, confidence, hit_count)
                    render_verdict(prediction, confidence, hit_count, title, text, is_domain_blocked)

                except ValueError as e:
                    st.error(str(e))
                except Exception as e:
                    st.error(f"The system has encountered an unforeseen disruption: {str(e)}")

    # ---- HISTORY TAB ----
    with tab_history:
        st.markdown(
            "<h3 style='font-family:Georgia,serif; color:#C25953;'>Inspection Registry</h3>",
            unsafe_allow_html=True
        )
        render_history(db)


if __name__ == "__main__":
    main()
