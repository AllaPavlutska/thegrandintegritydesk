class DatabaseManager:
    """Handles all SQLite database connections and queries."""
    
    def __init__(self, db_name="safety_pipeline.db"):
        self.db_name = db_name
        self._initialize_db()

    def _initialize_db(self):
        """Creates the database table if it doesn't exist."""
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS detections (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    url TEXT NOT NULL,
                    article_title TEXT,
                    is_misinfo INTEGER,
                    confidence_score REAL,
                    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
                )
            ''')
            conn.commit()

    def log_detection(self, url, title, is_misinfo, confidence):
        """Saves a new analysis result into the database."""
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.cursor()
            cursor.execute('''
                INSERT INTO detections (url, article_title, is_misinfo, confidence_score)
                VALUES (?, ?, ?, ?)
            ''', (url, title, int(is_misinfo), float(confidence)))
            conn.commit()

class MisinformationEngine:
    """Handles text extraction, domain rules, and ML classification."""
    
    def __init__(self):
 
        self.blacklisted_domains = [
            '.ru/',             # Blanket ban on all Russian Top-Level Domains
            'rt.com',           # Russia Today (State Media)
            'sputniknews.com',  # Sputnik (State Media)
            'tass.com',         # TASS News Agency
            'rbth.com'          # Russia Beyond
        ]
        
  
        self.vectorizer = TfidfVectorizer(stop_words='english')
        self.model = LogisticRegression()
        self._train_micro_model()

    def _train_micro_model(self):
        """Trains a real ML model on a small dataset."""
        training_data = [
            ("Shocking secret cure doctors don't want you to know about!", 1),
            ("The government is hiding aliens in a secret bunker, act now!", 1),
            ("Guaranteed crypto returns, send money to this wallet today.", 1),
            ("The central bank announced a 0.5% interest rate hike today.", 0),
            ("Researchers published a new peer-reviewed study on climate change.", 0),
            ("City council votes to approve the new public park funding.", 0)
        ]
        texts = [item[0] for item in training_data]
        labels = [item[1] for item in training_data]
        
        X_vectors = self.vectorizer.fit_transform(texts)
        self.model.fit(X_vectors, labels)

    def check_domain_rules(self, url):
        """Checks if the URL triggers our hardcoded blacklist."""
        url_lower = url.lower()
        for domain in self.blacklisted_domains:
            if domain in url_lower:
                return True
        return False

    def extract_text(self, url):
        """Scrapes the text and title from a given URL."""
        try:
            article = Article(url)
            article.download()
            article.parse()
            return article.title, article.text
        except Exception as e:
            raise ValueError("Could not extract content. Please ensure the URL is valid and publicly accessible.")

    def analyze(self, text):
        """Runs the text through the ML model and returns prediction & confidence."""
        if not text.strip():
            raise ValueError("The article appears to be empty or blocked by a paywall.")
            
        vectorized_text = self.vectorizer.transform([text])
        prediction = self.model.predict(vectorized_text)[0]
        probabilities = self.model.predict_proba(vectorized_text)[0]
        
        confidence = probabilities[prediction] 
        return prediction, confidence
